\chapter{\Etale\ Site}


\begin{definition}[\Etale\ covering]
	(\cite[][(Definition 20.21), p.~101]{GoertzWedhornAG2})
	An \etale\ covering of a scheme $X$ is a family of \etale\ morphisms $(g_i: U_i\to X)_{i\in I}$, such that $X = \bigcup_{i\in I} g_i(U_i)$. Let $\mathcal{U} =(U_i\to X)_{i\in I}$ and $\mathcal{V} = (V_j\to X)_{j\in J}$ be \etale\ coverings, than we call $\mathcal{V}$ a refinement of $\mathcal{U}$, if there exists a map $\alpha: J\to I$, such that for each $j\in J$ there exists a morphism $V_j\to U_{\alpha(j)}$, such that the following commute:
	% https://q.uiver.app/#q=WzAsMyxbMCwwLCJWX2oiXSxbMiwwLCJVX3tcXGFscGhhX2p9Il0sWzEsMSwiWCJdLFswLDJdLFsxLDJdLFswLDEsIlxcdmFycGhpX2oiXV0=
	\[\begin{tikzcd}
		{V_j} && {U_{\alpha(j)}} \\
		& X
		\arrow[from=1-1, to=1-3]
		\arrow[from=1-1, to=2-2]
		\arrow[from=1-3, to=2-2]
	\end{tikzcd}\]
\end{definition}

\begin{proposition}\label{prop:affine-etale-refinement}
	(\cite[][p.~101]{GoertzWedhornAG2})
	Let $X$ be a scheme. Then every \etale\ covering $\mathcal{U} =(g_i: U_i\to X)_{i\in I}$ of $X$ admits a refinement $\mathcal{V} = (V_j\to X)_{j\in J}$ of affine schemes. If $X$ quasi-compact, the refinement can be chosen to be finite.
\end{proposition}

\begin{proof}
	For each $(U_i)_{i\in I}$ chose an open affine cover $U_i = \bigcup_{j\in J_i} V_{ij}$. Now consider the composite $V_{ij} \hookrightarrow U_i\xrightarrow{g_i} X$ . Each $V_{ij}\hookrightarrow U_i$ is an open immersion, hence \etale. Let $J = \coprod_{i\in I}J_i = \{(i, j) \mid i\in I, j\in J_i\}$.  This yields an \etale\ cover $\mathcal{V} = (V_{ij} \to X)_{(i, j)\in J}$, since the composite of two \etale\ morohisms is \etale\ and the image covers $X$ by construction.
	$\mathcal{V}$ is a refinement of $\mathcal{U}$ via $\alpha: J\to I, (i, j) \mapsto i$.
	If $X$ is quasi-compact, there exists a finite subset $J'\subseteq J$ such that the opens $g_{\alpha(j)}(V_j)$ cover $X$.
\end{proof}

This allows us to restrict to finite affine \etale\ covers. Such reductions are standard in
algebraic geometry: many statements can then be checked on affines and translated into
commutative algebra.

\begin{theorem}[Representable presheaves of schemes]
	Let $X$ be a scheme. Consider the representable presheaf 
	\[h_X: (\SchOver{Et})^\op \to \Set\]
	on the \etale\ site of \Sch. Then $h_X$ is a sheaf in the \etale\ topology.
\end{theorem}

\begin{proof}
	Let $U$ be a scheme and $\mathcal{U} = (U_i\to U)_{i\in I}$ be an \etale\ covering. We need to show, that the following diagram is exact:
	% https://q.uiver.app/#q=WzAsMyxbMCwwLCJoX1goVSkiXSxbMSwwLCJcXHByb2RfaSBoX1goVV9pKSJdLFsyLDAsIlxccHJvZF97aSxqfSBoX1goVV9pXFx0aW1lc19VIFVfaikiXSxbMCwxXSxbMSwyLCIiLDAseyJvZmZzZXQiOjJ9XSxbMSwyLCIiLDAseyJvZmZzZXQiOi0yfV1d
	\[\begin{tikzcd}
		{h_X(U)} & {\prod_i h_X(U_i)} & {\prod_{i,j} h_X(U_i\times_U U_j)}
		\arrow[from=1-1, to=1-2]
		\arrow[shift right=2, from=1-2, to=1-3]
		\arrow[shift left=2, from=1-2, to=1-3]
	\end{tikzcd}\]
	Obviously the composite is equal by the pullback property.
	
	We will first reduce $U$ to the affine case.
	Let $U = \bigcup_\alpha V_\alpha$ be an affine open cover.  Assume $f_i:U_i\to X$, such that $U_i\times_U U_j\to U_i \xrightarrow{f_i} X$ and $U_i\times_U U_j\to U_j \xrightarrow{f_j} X$ coincide. To construct a unique morphism $f:U\to X$ it suffices to construct compatible morphisms $g_\alpha:V_\alpha \to X$ for all $\alpha$.
	
	
	Consider the following pullback diagram.

	\[\begin{tikzcd}
		{V_\alpha\times_UU_i} & {U_i} \\
		{V_\alpha} & U
		\arrow[from=1-1, to=1-2]
		\arrow[from=1-1, to=2-1]
		\arrow[from=1-2, to=2-2]
		\arrow[from=2-1, to=2-2]
		\arrow[phantom, from=1-1, to=2-2, "\lrcorner", pos=0.15]
	\end{tikzcd}\]
	The morphisms $V_\alpha\times_UU_i \to V_\alpha$ are \etale\ by proposition \ref{prop:etale-stable-under-pullback}. The morphism $V_\alpha\hookrightarrow U$ is the inclusion of the affine open $V_\alpha$, which is an open immersion, hence \etale. Write $V_{i, \alpha} := V_\alpha\times_UU_i$.
	Assume the sheaf condition holds for $(V_{i, \alpha}\to V_\alpha)_{i\in I}$ for all $\alpha$, i.e. the following diagram is a exact:
	% https://q.uiver.app/#q=WzAsMyxbMCwwLCJoX1goVl9cXGFscGhhKSJdLFsxLDAsIlxccHJvZF9pIGhfWChWX3tpLFxcYWxwaGF9KSJdLFsyLDAsIlxccHJvZF97aSxqfWhfWChWX3tpLFxcYWxwaGF9XFx0aW1lc197Vl9cXGFscGhhfSBWX3tqLFxcYWxwaGF9KSJdLFswLDFdLFsxLDIsIiIsMCx7Im9mZnNldCI6Mn1dLFsxLDIsIiIsMCx7Im9mZnNldCI6LTJ9XV0=
	\[\begin{tikzcd}
		{h_X(V_\alpha)} & {\prod_i h_X(V_{i,\alpha})} & {\prod_{i,j}h_X(V_{i,\alpha}\times_{V_\alpha} V_{j,\alpha})}
		\arrow[from=1-1, to=1-2]
		\arrow[shift right=2, from=1-2, to=1-3]
		\arrow[shift left=2, from=1-2, to=1-3]
	\end{tikzcd}\]
	Write $f_{i, \alpha}: V_{i, \alpha} \to U_i \xrightarrow{f_i} X$ for the restriction of $f_i$ to $V_{i, \alpha}$. Since $f_i$ and $f_j$ agree on $U_i\times_U U_j$, the following commutes: 
	% https://q.uiver.app/#q=WzAsNCxbMCwwLCJVX2lcXHRpbWVzX1UgVV9qIl0sWzEsMCwiVV9qIl0sWzAsMSwiVV9pIl0sWzEsMSwiWCJdLFsxLDMsImZfaiJdLFsyLDMsImZfaSIsMl0sWzAsMl0sWzAsMV1d
	\[\begin{tikzcd}
		{U_i\times_U U_j} & {U_j} \\
		{U_i} & X
		\arrow[from=1-1, to=1-2]
		\arrow[from=1-1, to=2-1]
		\arrow["{f_j}", from=1-2, to=2-2]
		\arrow["{f_i}"', from=2-1, to=2-2]
	\end{tikzcd}\]

	By functoriality of $V_\alpha\times_U -$ we obtain a commutative diagram:

	\[\begin{tikzcd}
		{(V_\alpha\times_U U_i)\times_{V_\alpha}(V_\alpha\times_U U_j)} & {V_\alpha\times_U(U_i\times_U U_j)} & {V_\alpha \times_U U_j} \\

	Then the compatible $f_{i, \alpha}$ glue uniquely to a morphism $g_\alpha:V_\alpha\to X$ fitting into the following diagramm:

	\[\begin{tikzcd}
		{V_{i, \alpha}\times_{V_\alpha}V_{j, \alpha}} & {V_{j, \alpha}} \\
		{V_{i, \alpha}} & {V_\alpha} \\

		\arrow[from=1-1, to=1-2]
		\arrow[from=1-1, to=2-1]
		\arrow[from=1-2, to=2-2]

	Next we prove, that the $g_\alpha$ agree on overlaps $V_{\alpha\beta} := V_\alpha \cap V_\beta \cong V_\alpha\times_U V_\beta$. Again by proposition \ref{prop:etale-stable-under-pullback} the following pullback is \etale:

	\[\begin{tikzcd}
		V_{i, \alpha\beta} & {U_i} \\
		{V_{\alpha\beta}} & U
		\arrow[from=1-1, to=1-2]

	\[\begin{tikzcd}
		{V_{i,\alpha\beta}} & {V_{i, \beta}} \\

	We see, that on the \etale\ cover $V_{i, \alpha\beta} \to V_{\alpha\beta}$ both restrictions of $g_\alpha$ and $g_\beta$ equal the same morphism, namely $f_i$ (or the restriction to $V_{i, \alpha\beta}$, to be precise).
	
	So all $g_\alpha$ can be glued together to a unique morphism $f:U\to X$, since $h_X$ is a sheaf in the Zariski site.
	Finally we need to check, that $f$ restricts to $f_i$ on $U_i$. This can be checked on all the $V_{i,\alpha}\to U_i$ for all $\alpha$. But this holds by construction.
	
	
	
\end{proof}
